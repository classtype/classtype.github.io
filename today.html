Вернуться
--------------------------------------------------------------------------------

# Быстрый коммит на сервер
git add .&&git commit --amend --no-edit&&git push -f

# Наглядно сравнить две ветки
git log master...feature --oneline --graph --boundary

--------------------------------------------------------------------------------
# Конфликты
--------------------------------------------------------------------------------

# Отменить слияние
git merge --abort

# Продолжить слияние
git merge --continue

# Добавить в конфликтный файл третью версию (до изменений)
git checkout --conflict=diff3 --merge index.html

# Сохранить в конфиг
git config --global merge.conflictStyle diff3

# Удалить из конфига
git config --global --unset merge.conflictStyle
