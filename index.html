#---------------------------------------------------------------------------------------------------
# Первый старт
#---------------------------------------------------------------------------------------------------

# Создать новый репозиторий
git init

# Добавить в индекс файл README.md
git add README.md

# Создать коммит с комметарием "first commit"
git commit -m "first commit"

# Привязать текущий репозиторий к удаленному "https://github.com/stanruss/название.git"
git remote add origin https://github.com/stanruss/название.git

# 
git push -u origin master

#---------------------------------------------------------------------------------------------------
# Работа с индексом
#---------------------------------------------------------------------------------------------------

git add index.html | Добавить в индекс файл index.html
git add . | Добавить в индекс все папки и файлы из текущей директории
git add --all | Добавить в индекс все папки и файлы из текущей директории и из всех поддиректориев
git add -A | Тоже самое, что и выше
