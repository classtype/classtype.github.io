git init
git add README.md | Добавить в индекс файл README.md
git commit -m "first commit" | Создать коммит с комметарием "first commit"
git remote add origin https://github.com/classtype/project.git
git push -u origin master

git status | Показать статус.
git clone https://github.com/classtype/project.git ~/project | Клонировать репозиторий в директорию project
git remote add origin https://github.com/classtype/project.git | Привязать удаленный репозиторий "origin" к локальному.
git remote rm origin | Отвязать удаленный репозиторий "origin" от локального.
git push -u origin master | Связать удаленный репозиторий "origin" с локальной веткой "master".
git pull | Скачать изменения с удаленного репозитория.

#-------------------------------------------------------------------------------
# Буфер обмена
#-------------------------------------------------------------------------------

git stash | Сохранить изменения в бумер обмена.
git stash pop | Выгрузить изменения сделаные предыдущей командой.

#-------------------------------------------------------------------------------
!!Доделать
git checkout @ . | Откатить все файлы до последнего коммита в текущей ветке, также сбросить индекс.
git checkout @ index.html | Тоже самое, но для файла "index.html", также сбросить файл "index.html" в индекс.
git checkout 6c2e81b index.html | Откатить файл "index.html" из коммита "6c2e81b".
git checkout master index.html | Откатить файл "index.html" из последнего коммита в ветке master.

git checkout 6c2e81b . | Откатить все файлы до коммита "6c2e81b" с добавлением в индекс.

#-------------------------------------------------------------------------------
# Рабочая директория / Индекс
#-------------------------------------------------------------------------------

git checkout @ . | Сбросить индекс и откатить все изменения в рабочей директории.
git checkout @ index.html | Тоже самое, но для файла "index.html".
git checkout . | Откатить все изменения до версии в индекс, если индекс пуст, то до версии коммита.
git checkout index.html | Тоже самое, но для файла "index.html".
git checkout 6c2e81b index.html | Тоже самое, но взять версию файла из коммита "6c2e81b".
git checkout 6c2e81b . | Тоже самое, но взять версию файла из коммита "6c2e81b".

#-------------------------------------------------------------------------------
# Индекс
#-------------------------------------------------------------------------------

git reset | Сбросить все файлы.
git reset index.html | Сбросить файл "index.html".

#-------------------------------------------------------------------------------
# Переместить указатель в текущей ветке на другой коммит
#-------------------------------------------------------------------------------

git reset --hard 6c2e81b | С удалением всех изменений в рабочей директории и в индекс.
git reset --soft 6c2e81b | С сохранением текущих изменений в рабочей директории и в индекс.
git reset 6c2e81b | С сохранением текущих изменений только в рабочей директории.
git reset --mixed 6c2e81b | Тоже самое, так как --mixed это флаг по умолчанию.

#-------------------------------------------------------------------------------
# Индексы
#-------------------------------------------------------------------------------

git add index.html | Добавить в индекс файл "index.html".
git add . | Добавить в индекс все папки и файлы из текущей директории.
git add --all | Добавить в индекс все папки и файлы из текущей директории и из всех поддиректориев.
git add -A | Тоже самое, только сокращенный вариант.

#-------------------------------------------------------------------------------
# Коммиты
#-------------------------------------------------------------------------------

git commit --amend -m "New comment" | Перезаписать последний коммит с комментарием "New comment".
git commit --amend --no-edit | Тоже самое, но можно заново не писать комментарий.
git commit -a -m "Comment" | Сделать коммит в обход git add, но только для файлов которые уже есть в репозитории.

#-------------------------------------------------------------------------------
# Список всех коммитов
#-------------------------------------------------------------------------------

git log | Просмотра списка всех коммитов.
git log --oneline | Каждый коммит показывается в одну строчку.

#-------------------------------------------------------------------------------
# Изменения в коммите
#-------------------------------------------------------------------------------

git show | Показать изменения в последнем коммите.
git show --name-only | Тоже самое, но показать только имена измененных файлов.
git show 6c2e81b | Просмотреть изменения по конкретному коммиту.
git show master | Просмотреть изменения по последнему коммиту в ветке master.
