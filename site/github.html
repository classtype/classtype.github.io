# Быстрый старт для текущего проекта
--------------------------------------------------------------------------------

git init
git add .
git commit -m '1.0.0'
git remote add origin git@classtype:classtype/PROJECT.git
git push -u origin master

--------------------------------------------------------------------------------
# Связать удаленный репозиторий с локальным
--------------------------------------------------------------------------------

#  Создать директорию "project"
mkdir project

# Перейти в директорию "project"
cd project

# Клонировать удаленный репозиторий в текущую директорию
git clone git@github.com:USER/PROJECT.git .

# Связать удаленный репозиторий "origin" с локальной веткой "master".
git push -u origin master

--------------------------------------------------------------------------------
# Общая информация
--------------------------------------------------------------------------------

git status | Показать статус.
git pull | Скачать изменения с удаленного репозитория.
git remote add origin git@github.com:USER/PROJECT.git | Привязать удаленный репозиторий "origin" к локальному.
git remote rm origin | Отвязать удаленный репозиторий "origin" от локального.
